local_repo ()

An optional local repository which defaults to ./lib/boson or ./.boson.

[show source] 
# File lib/boson.rb, line 35
  def local_repo
    @local_repo ||= begin
      dir = ["lib/boson", ".boson"].find {|e| File.directory?(e) &&
         File.expand_path(e) != repo.dir }
      Repo.new(dir) if dir
    end
  end
